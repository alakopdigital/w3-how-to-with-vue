<script>
/**
 * IconBar Component
 *
 * 함수형 컴포넌트로 작성
 */
export default {
  name: 'icon-bar',
  functional: true,
  props: {
    // is-vertical일 경우 horizontal 대신 vertical class를 적용
    isVertical: Boolean
  },
  render(h, ctx) {
    const cx = ['icon-bar']
 
    cx.push(ctx.props.isVertical ? 'vertical' : 'horizontal')
 
    const cls = cx.reduce((cls, name) => {
      cls[name] = true
      return cls
    }, {})
 
    return h('div', {'class': cls}, ctx.children)
  }
}
</script>
